38  /**
39  * @brief Convert a VariableT to a std::string
40  * @param var - Input variable
41  * @param use_quotes_for_string - When true, variables with String type are returned with enclosing double quotation marks for file/human-readable output.
42  * @returns std::string
43  * @throws bad_variant_access when the given VariableT instance is null/std::monostate.
44  */
45  std::string to_string(const VariableT& var, const bool use_quotes_for_string = true)
46  {
47  if (std::holds_alternative<String>(var)) { // String
48  const auto str{ std::get<String>(var) };
49  return (use_quotes_for_string ? "\"" + str + "\"" : str);
50  }
51  else if (std::holds_alternative<Float>(var)) // Float
52  return std::to_string(std::get<Float>(var));
53  else if (std::holds_alternative<Integer>(var)) // Integer
54  return std::to_string(std::get<Integer>(var));
55  else if (std::holds_alternative<Boolean>(var)) // Boolean
56  return str::bool_to_string(std::get<Boolean>(var));
57  throw std::bad_variant_access();
58  }

119  /**
120  * @brief Stream insertion operator. Outputs in valid file format, with quotation-enclosed strings.
121  * @param os - (implicit) Target output stream.
122  * @param obj - (implicit) Target INIContainer instance.
123  * @returns std::ostream&
124  */
125  friend std::ostream& operator<<(std::ostream& os, const INIContainer& obj)
126  {
127  using enum AlignType;
128  std::streamsize align_width{ 0ll };
129  switch (obj._align_output) {
130  case SECTION:
131  for (const auto& [header, section] : obj._cont) {
132  if (!header.empty())
133  os << '[' << header << ']' << '\n';
134  const auto align_width{ [&section]() -> std::streamsize {
135  size_t longest{ 0ll };
136  for (auto& [keyname, _] : section)
137  if (const auto sz{ keyname.size() }; sz > longest)
138  longest = sz;
139  return static_cast<std::streamsize>(longest);
140  }() };
141  for (auto& [key, val] : section)
142  os << key << str::VIndent(align_width + 2ll, key.size()) << "= " << to_string(val) << '\n';
143  os << '\n';
144  }
145  break;
146  case ALL:
147  align_width = [&obj]() -> std::streamsize {
148  size_t longest{ 0ll };
149  for (auto& [_, section] : obj._cont)
150  for (auto& [keyname, __] : section)
151  if (const auto sz{ keyname.size() }; sz > longest)
152  longest = sz;
153  return static_cast<std::streamsize>(longest);
154  }();
155  [[fallthrough]];
156  case NONE:
157  for (auto& [header, section] : obj._cont) {
158  os << '[' << header << ']' << '\n';
159  for (auto& [key, value] : section)
160  os << key << str::VIndent(align_width + 2ll, key.size()) << "= " << to_string(value, true) << '\n';
161  os << '\n';
162  }
163  break;
164  }
165  return os;
166  }
