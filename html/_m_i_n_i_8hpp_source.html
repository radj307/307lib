194  /// @brief Lambda that handles variable verification & insertion into the map.
195  const auto& insert{ [&]() {
196  if (!key.has_value())
197  throwEx(ln, "Missing Key Declaration");
198  if (!value.has_value()) {
199  if (allowBlankValue)
200  value = std::string{};
201  else
202  throwEx(ln, "Missing Value Declaration");
203  }

205  map.insert(std::make_pair(header, inner_container{}));
206  map[header].insert_or_assign(key.value(), value.value());
207  } };

210 
211  size_t i{ 0ull };

213  ++i; // increment index by one (used to add _EOL)
214  switch (type) {
215  case TokenType::HEADER: // set the header
216  header = str;
217  break;
218  case TokenType::SETTER: // set the temp setter variable to true if it isn't already
219  if (!setter)
220  setter = true;
221  else throwEx(ln, "Duplicate Setters"); // throw duplicate exception if setter was already true
222  break;
223  case TokenType::KEY: // set the temp key
224  if (!key.has_value()) {
225  key = str;
226  break;
227  }
228  else [[fallthrough]]; // if the key is already set, parse as a string value
229  case TokenType::BOOLEAN: [[fallthrough]];
230  case TokenType::NUMBER: [[fallthrough]]; // set the temp value to a long double
231  case TokenType::NUMBER_INT: [[fallthrough]];
232  case TokenType::STRING: // set the temp value to a string
233  if (!setter)
234  throwEx(ln, "Missing Setter");
235  if (value.has_value())
236  value = value.value() + str;
237  else
238  value = str;
239  break;
240  case TokenType::NEWLINE: // insert & reset temp variables
241  if (any_defined())
242  insert();
243  key = std::nullopt;
244  value = std::nullopt;
245  setter = false;
246  ++ln;
247  break;
248  case TokenType::END: // insert temp variables and return successfully.
249  if (any_defined())
250  insert();
251  if (i == tokens.size()) // if this EOF is the last element in the tokens vec.
252  return map;
253  break;
254  case TokenType::COMMENT:break;

257  }
