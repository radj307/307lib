501  struct INITokenizer : token::base::TokenizerBase<LEXEME, LexemeDict, ::token::DefaultDefs::TokenType, Token> {
502  INITokenizer(std::stringstream&& buffer) : TokenizerBase(std::move(buffer), LEXEME::WHITESPACE) {}
503  protected:
504  TokenT getNext() override
505  {
506  const auto ch{ getch() };
507  switch (get_lexeme(ch)) {
508  case LEXEME::POUND: [[fallthrough]];
509  case LEXEME::SEMICOLON:// Comment
510  return Token{ std::string(1ull, ch) += getline('\n', false), TokenType::COMMENT };
511  case LEXEME::EQUALS: // Setter Start
512  return Token{ ch, TokenType::SETTER };
513  case LEXEME::QUOTE_SINGLE: // String (single) start
514  return Token{ getline('\''), TokenType::STRING };
515  case LEXEME::QUOTE_DOUBLE: // String (double) start
516  return Token{ getline('\"'), TokenType::STRING };
517  case LEXEME::SQUAREBRACKET_OPEN:
518  return Token{ getline(']'), TokenType::HEADER };
519  case LEXEME::LETTER_LOWERCASE: [[fallthrough]]; // if text appears without an enclosing quote, parse it as a key
520  case LEXEME::LETTER_UPPERCASE:
521  { // the case of unenclosed string variables is handled by the parser, not the tokenizer
522  const auto pos{ rollback() };
523  std::string str;
524  if (const auto lc{ str::tolower(ch) }; lc == 't' && getline_and_match(4u, "true", str))
525  return{ str, TokenType::BOOLEAN };
526  else if (lc == 'f' && getline_and_match(5u, "false", str))
527  return{ str, TokenType::BOOLEAN };
528  else
529  ss.seekg(pos - 1ll);
530  // getnotsimilar to newlines, equals, pound/semicolon '#'/';' (comments), and EOF
531  return Token{ str::strip_line(getnotsimilar(LEXEME::NEWLINE, LEXEME::EQUALS, LEXEME::POUND, LEXEME::SEMICOLON, LEXEME::_EOF), "#;"), TokenType::KEY };
532  }
533  case LEXEME::SUBTRACT: [[fallthrough]]; // number start
534  case LEXEME::DIGIT:
535  {
536  rollback();
537  const auto str{ getsimilar(LEXEME::SUBTRACT, LEXEME::DIGIT, LEXEME::PERIOD) };
538  return (str::pos_valid(str.find('.') || !str.empty() && (str.back() == 'f' || str.back() == 'F')) ? Token{ str, TokenType::NUMBER } : Token{ str, TokenType::NUMBER_INT });
539  }
540  case LEXEME::NEWLINE:
541  return Token{ ch, TokenType::NEWLINE };
542  case LEXEME::WHITESPACE:
543  throw make_exception("TokenizerINI::getNext()\tReceived unexpected whitespace character as input!");
544  case LEXEME::ESCAPE:
545  return Token{ std::string(1u, ch) += getch(true), TokenType::ESCAPED };
546  case LEXEME::_EOF:
547  return Token{ TokenType::END };
548  default:
549  return Token{ ch, TokenType::NULL_TYPE };
550  }
551  }

575  OutputT parse() const
576  {
577  // Init temporary variables:
578  std::string header{ "" }; // ini standard allows empty headers
579  std::optional<std::string> key;
580  std::optional<VariableT> value;
581  bool setter{ false };
582  size_t ln{ 1ull };
583  // Init map:
584  OutputT map{};
585 
586  strip_types(TokenType::COMMENT); // remove all comments
587 
588  /// @brief Lambda for throwing exceptions with consistent naming conventions.
589  /// @brief Lambda that handles variable verification & insertion into the map.
590  const auto& insert{ [&]() {
591  if (!key.has_value())
592  throwEx(ln, "Missing Key Declaration");
593  if (!value.has_value()) {
594  if (allowBlankValue)
595  value = std::string{};
596  else
597  throwEx(ln, "Missing Value Declaration");
598  }
599  map[header].insert_or_assign(key.value(), value.value());
600  } };
601  /// @brief Lambda that returns true if any temp variable is defined. This is used to detect whether a NEWLINE occurrance is valid or not
602  const auto& any_defined{ [&key, &value, &setter]() { return key.has_value() || value.has_value() || setter; } };
603 
604  size_t i{ 0ull };
605  for (const auto& [type, str] : tokens) {
606  ++i; // increment index by one (used to add _EOL)
607  switch (type) {
608  case TokenType::HEADER: // set the header
609  header = str;
610  break;
611  case TokenType::SETTER: // set the temp setter variable to true if it isn't already
612  if (!setter)
613  setter = true;
614  else throwEx(ln, "Duplicate Setters"); // throw duplicate exception if setter was already true
615  break;
616  case TokenType::KEY: // set the temp key
617  if (!key.has_value()) {
618  key = str;
619  break;
620  }
621  else [[fallthrough]]; // if the key is already set, parse as a string value
622  case TokenType::STRING: // set the temp value to a string
623  if (!setter)
624  throwEx(ln, "Missing Setter");
625  if (value.has_value()) {
626  if (std::holds_alternative<String>(value.value()))
627  value = (std::get<String>(value.value()) + str); // allow fallthrough "keys" (additional values that were cut off by the tokenizer -- shouldn't happen but just in case) to be appended
628  else throwEx(ln, "Duplicate Value");
629  }
630  else
631  value = str;
632  break;
633  case TokenType::NUMBER: // set the temp value to a long double
634  if (!setter)
635  throwEx(ln, "Missing Setter");
636  if (size_t decimal_count{ 0ull }; std::all_of(str.begin(), str.end(), [&decimal_count](auto&& ch) { if (ch == '.') ++decimal_count; return isdigit(ch) && decimal_count == 1; }))
637  value = str::stold(str);
638  else value = str;
639  break;
640  case TokenType::NUMBER_INT: // set the temp value to an integer
641  if (!setter)
642  throwEx(ln, "Missing Setter");
643  if (std::all_of(str.begin(), str.end(), isdigit))
644  value = str::stoll(str);
645  else value = str;
646  break;
647  case TokenType::BOOLEAN:
648  { // set the temp value to a boolean
649  if (!setter)
650  throwEx(ln, "Missing Setter");
651  auto b{ str::string_to_bool(str) };
652  if (!b.has_value())
653  value = str;
654  else
655  value = std::move(b);
656  break;
657  }
658  case TokenType::NEWLINE: // insert & reset temp variables
659  if (any_defined())
660  insert();
661  key = std::nullopt;
662  value = std::nullopt;
663  setter = false;
664  ++ln;
665  break;
666  case TokenType::END: // insert temp variables and return successfully.
667  if (any_defined())
668  insert();
669  if (i == tokens.size()) // if this EOF is the last element in the tokens vec.
670  return map;
671  break;
672  default: // throw unexpected token exception
673  throwEx(ln, "Unexpected token type");
674  }
675  }
676  // throw missing EOF exception
677  throwEx(ln, "Missing Token: END / EOF");
678  return{}; // this will never be reached, but it prevents compiler warning
679  }
680  };
681  }
