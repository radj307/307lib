44  /**
45  * @brief Convert a VariableT to a std::string
46  * @param var - Input variable
47  * @param use_quotes_for_string - When true, variables with String type are returned with enclosing double quotation marks for file/human-readable output.
48  * @returns std::string
49  * @throws bad_variant_access when the given VariableT instance is null/std::monostate.
50  */
51  std::string to_string(const VariableT& var, const bool& use_quotes_for_string = false)
52  {
53  if (std::holds_alternative<String>(var)) { // String
54  const auto str{ std::get<String>(var) };
55  return (use_quotes_for_string ? "\"" + str + "\"" : str);
56  }
57  else if (std::holds_alternative<Float>(var)) // Float
58  return std::to_string(std::get<Float>(var));
59  else if (std::holds_alternative<Integer>(var)) // Integer
60  return std::to_string(std::get<Integer>(var));
61  else if (std::holds_alternative<Boolean>(var)) // Boolean
62  return str::bool_to_string(std::get<Boolean>(var));
63  throw std::bad_variant_access();
64  }

162  /**
163  * @brief Stream insertion operator. Outputs in valid file format, with quotation-enclosed strings.
164  * @param os (implicit) Target output stream.
165  * @param obj (implicit) Target INIContainer instance.
166  * @returns std::ostream&
167  */
168  friend std::ostream& operator<<(std::ostream& os, const INIContainer& obj)
169  {
170  std::streamsize align_width{ 0ll };
171  switch (obj._align_output) {
172  case AlignType::SECTION:
173  for (const auto& [header, section] : obj._cont) {
174  if (!header.empty())
175  os << '[' << header << ']' << '\n';
176  const auto align_width{ [&section]() -> std::streamsize {
177  size_t longest{ 0ll };
178  for (auto& [keyname, _] : section)
179  if (const auto sz{ keyname.size() }; sz > longest)
180  longest = sz;
181  return static_cast<std::streamsize>(longest);
182  }() };
183  for (auto& [key, val] : section)
184  os << key << shared::indent(align_width + 2ll, key.size()) << "= " << to_string(val) << '\n';
185  os << '\n';
186  }
187  break;
188  case AlignType::ALL:
189  align_width = [&obj]() -> std::streamsize {
190  size_t longest{ 0ll };
191  for (auto& [_, section] : obj._cont)
192  for (auto& [keyname, __] : section)
193  if (const auto sz{ keyname.size() }; sz > longest)
194  longest = sz;
195  return static_cast<std::streamsize>(longest);
196  }();
197  [[fallthrough]];
198  case AlignType::NONE:
199  for (auto& [header, section] : obj._cont) {
200  os << '[' << header << ']' << '\n';
201  for (auto& [key, value] : section)
202  os << key << shared::indent(align_width + 2ll, key.size()) << "= " << to_string(value, true) << '\n';
203  os << '\n';
204  }
205  break;
206  }
207  return os;
208  }

582  * @struct INIParser
583  * @brief Token parser for the INI file format.
584  */
585  struct INIParser : token::base::TokenParserBase<INIContainer::Map, Token> {
586  bool allowBlankValue{ true };
587  const std::string filename{};
588 
589 # if LANG_CPP >= 17
590  INIParser(std::vector<Token>&& tokens, const std::filesystem::path& filename = {}) : TokenParserBase(std::move(tokens)), filename{ filename.generic_string() } {}
591 # else
592  INIParser(std::vector<Token>&& tokens, const std::string_view& filename = {}) : TokenParserBase(std::move(tokens)), filename{ filename } {}
593 # endif
594 
595  private:
596  void throwEx(const size_t& line, const std::string& msg) const
597  {
598  throw make_exception("Syntax Error: ", msg, " at line ", line, " in file: \"", filename, '\"');
599  }
600  public:
601 
602  OutputT parse() const
603  {
604  // Init temporary variables:
605  std::string header{ "" }; // ini standard allows empty headers
606  std::optional<std::string> key;
607  std::optional<VariableT> value;
608  bool setter{ false };
609  size_t ln{ 1ull };
610  // Init map:
611  OutputT map{};
612 
613  strip_types(TokenType::COMMENT); // remove all comments
614 
615  /// @brief Lambda for throwing exceptions with consistent naming conventions.
616  /// @brief Lambda that handles variable verification & insertion into the map.
617  const auto& insert{ [&]() {
618  if (!key.has_value())
619  throwEx(ln, "Missing Key Declaration");
620  if (!value.has_value()) {
621  if (allowBlankValue)
622  value = std::string{};
623  else
624  throwEx(ln, "Missing Value Declaration");
625  }
626  map[header].insert_or_assign(key.value(), value.value());
627  } };
628  /// @brief Lambda that returns true if any temp variable is defined. This is used to detect whether a NEWLINE occurrance is valid or not
629  const auto& any_defined{ [&key, &value, &setter]() { return key.has_value() || value.has_value() || setter; } };
630 
631  size_t i{ 0ull };
632  for (const auto& [type, str] : tokens) {
633  ++i; // increment index by one (used to add _EOL)
634  switch (type) {
635  case TokenType::HEADER: // set the header
636  header = str;
637  break;
638  case TokenType::SETTER: // set the temp setter variable to true if it isn't already
639  if (!setter)
640  setter = true;
641  else throwEx(ln, "Duplicate Setters"); // throw duplicate exception if setter was already true
642  break;
643  case TokenType::KEY: // set the temp key
644  if (!key.has_value()) {
645  key = str;
646  break;
647  }
648  else [[fallthrough]]; // if the key is already set, parse as a string value
649  case TokenType::STRING: // set the temp value to a string
650  if (!setter)
651  throwEx(ln, "Missing Setter");
652  if (value.has_value()) {
653  if (std::holds_alternative<String>(value.value()))
654  value = (std::get<String>(value.value()) + str); // allow fallthrough "keys" (additional values that were cut off by the tokenizer -- shouldn't happen but just in case) to be appended
655  else throwEx(ln, "Duplicate Value");
656  }
657  else
658  value = str;
659  break;
660  case TokenType::NUMBER: // set the temp value to a long double
661  if (!setter)
662  throwEx(ln, "Missing Setter");
663  if (size_t decimal_count{ 0ull }; std::all_of(str.begin(), str.end(), [&decimal_count](auto&& ch) { if (ch == '.') ++decimal_count; return isdigit(ch) && decimal_count == 1; }))
664  value = str::stold(str);
665  else value = str;
666  break;
667  case TokenType::NUMBER_INT: // set the temp value to an integer
668  if (!setter)
669  throwEx(ln, "Missing Setter");
670  if (std::all_of(str.begin(), str.end(), isdigit))
671  value = str::stoll(str);
672  else value = str;
673  break;
674  case TokenType::BOOLEAN:
675  { // set the temp value to a boolean
676  if (!setter)
677  throwEx(ln, "Missing Setter");
678  auto b{ str::string_to_bool(str) };
679  if (!b.has_value())
680  value = str;
681  else
682  value = std::move(b);
683  break;
684  }
685  case TokenType::NEWLINE: // insert & reset temp variables
686  if (any_defined())
687  insert();
688  key = std::nullopt;
689  value = std::nullopt;
690  setter = false;
691  ++ln;
692  break;
693  case TokenType::END: // insert temp variables and return successfully.
694  if (any_defined())
695  insert();
696  if (i == tokens.size()) // if this EOF is the last element in the tokens vec.
697  return map;
698  break;
699  case TokenType::COMMENT:break;
700  default: // throw unexpected token exception
701  //throwEx(ln, "Unexpected token type");
702  break;
703  }
704  }
705  // throw missing EOF exception
706  throwEx(ln, "Missing Token: END / EOF");
707  return{}; // this will never be reached, but it prevents compiler warning
708  }
709  };
710  }
