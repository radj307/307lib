540  /**
541  * @brief Parses the entire stringstream supplied in the XMLParser constructor.
542  * @returns An object of type T that represents the input XML data.
543  */
544  T parse() override
545  {
546  T root{};
547 
548  std::stack<XMLElement*> nodeStack;
549 
550  bool breakLoop{ false };
551  for (auto it{ this->tokens.begin() }; !breakLoop && it != this->tokens.end(); ++it) {
552  const auto& tkn{ *it };
553  const string_t s{ tkn.str };
554 
555  switch (tkn.type) {
556  case TokenType::ProcessingInstructions: {
557  if constexpr (std::same_as<T, XMLDocument>) {
558  if (!root.Instructions.attributes.empty())
559  throwex("parse", "syntax error", "Duplicate processing instruction definition '", tkn.str, '\'');
560 
561  const auto& matches{ parseTag(tkn.str) };
562 
563  root.Instructions.target = matches[1];
564  root.Instructions.attributes = parseAttributes(matches[2]);
565  } // else discard Instructions since there is nowhere to store them
566  break;
567  }
568  case TokenType::Tag: {
569  // parse name & attribute substrings:
570  const auto& matches{ parseTag(s) };
571 
572  const string_t name{ matches[1].str() };
573  attribute_container attributes;
574 
575  if (matches[2].matched)
576  attributes = parseAttributes(matches[2].str());
577 
578  // Determine the type of this tag
579  switch (getTagTypeOf(s)) {
580  case TagType::Open: {
581  if (nodeStack.size() == 0) { // this is the root node
582  root.name = name;
583  root.attributes = std::move(attributes);
584  nodeStack.push(&root);
585  }
586  else { // this isn't the root node
587 
588  std::visit([this, &nodeStack, &name, &attributes](auto&& value) {
589  using U = std::decay_t<decltype(value)>;
590 
591  if constexpr (std::same_as<U, null_t>)
592  nodeStack.top()->value = nodes_t{ XMLElement{ name, std::move(attributes) } };
593  else if constexpr (std::same_as<U, nodes_t>)
594  value.emplace_back(XMLElement{ name, std::move(attributes) });
595  else
596  throwex("parse", "syntax error", "Elements with value types cannot contain sub-elements!");
597  }, nodeStack.top()->value);
598 
599  if (auto* p = std::get_if<nodes_t>(&nodeStack.top()->value))
600  nodeStack.push(&p->back());
601  else throwex("parse", "logic error", "Failed to resolve value type!");
602  }
603  break;
604  }
605  case TagType::Close: {
606  auto top{ nodeStack.top() };
607  nodeStack.pop();
608 
609  if (top->name != name)
610  throwex("parse", "syntax error", "Missing closing tag for '", top->name, "' before '", name, '\'');
611  else if (nodeStack.size() == 0)
612  return root;
613  break;
614  }
615  case TagType::SelfClose:
616  if (nodeStack.size() == 0ull) {
617  root.name = name;
618  root.attributes = std::move(attributes);
619  if (std::distance(it, this->tokens.end()) <= 1ull)
620  throwex("parse", "syntax error", "Cannot parse multiple root elements!");
621  }
622  else {
623  std::visit([&s, &name, &attributes, &nodeStack](auto&& top) {
624  using U = std::decay_t<decltype(top)>;
625 
626  if constexpr (std::same_as<U, null_t>)
627  nodeStack.top()->value = nodes_t{ XMLElement{ name, attributes } };
628  else if constexpr (std::same_as<U, nodes_t>)
629  top.emplace_back(XMLElement{ name, attributes });
630  else throw make_exception("XMLParser::parse() syntax error: Illegal characters '", s, '\'');
631  }, nodeStack.top()->value);
632  }
633  break;
634  }
635  break;
636  }
637  case TokenType::Value: {
