30  /**
31  * @brief
32  * @tparam Pred
33  * @param begin
34  * @param end
35  * @param fn A function that accepts (IteratorT::value_type) and returns an integral or boolean.
36  * @returns size_t
37  */
38  template<class Pred, std::input_or_output_iterator IteratorT> static CONSTEXPR size_t count(IteratorT begin, IteratorT end, const Pred& fn) noexcept(false)
39  {
40  size_t counter{ 0ull };
41  for (auto it{ begin }; it != end; ++it) {
42  auto n{ fn(*it) };
43  using T = std::decay_t<decltype(n)>;
44  if constexpr (std::same_as<T, bool> || std::integral<T>)
45  counter += static_cast<size_t>(n);
46  }
47  return counter;
48  }

162  /**
163  * @concept streamable
164  * @brief Requires type T to define an `operator<<` that inserts into the specified stream type.
165  * @tparam T - A type to test for a valid operator<< overload.
166  * @tparam TStream - The type of input stream to test with. Defaults to std::ostream.
167  */
168  template<typename T, class OutputStream = std::ostream> concept streamable = requires (T inst)
169  {
170  // For a type to actually be streamable, operator<< must accept and return a
171  // reference to type OutputStream. However, checking the returned type from this
172  // compound requirement is inconsistent when the type doesn't *directly* define
173  // an `operator<<`, like is the case with most enums. (The operator implemented
174  // by the compiler doesn't always return the expected type.)
175  // What we're really looking for is that the operator can be chained, so instead
176  // we can just chain multiple "<< inst" to achieve the same result.
177  { std::declval<std::remove_cvref_t<OutputStream>&>() << inst << inst };
178  };
