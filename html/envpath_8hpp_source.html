27  /**
28  * @brief Gets the filepaths in the PATH environment variable.
29  * @detail You can use getvar() to get the PATH environment variable as a std::string.
30  * @returns A vector containing a filesystem::path for each entry in the PATH variable.
31  */
32  inline std::vector<std::filesystem::path> get_path()
33  {
34  std::vector<std::filesystem::path> paths;
35 
36  // retrieve the PATH variable
37  if (std::string result{ getvar(PATH_VAR_NAME).value_or("") };
38  !result.empty()) {
39  // reserve enough space for all of the paths
40  paths.reserve(std::count(result.begin(), result.end(), ';') + 1);
41 
42  // enumerate the result string and split by semicolons
43  for (auto it{ result.begin() }, it_end{ result.end() }, prev_begin{ it };
44  ; //< enter the loop even when at the end
45  ++it) {
46  if (it == it_end) {
47  // this is the last path, push it and break from the loop
48  paths.emplace_back(std::filesystem::path{ std::string_view(prev_begin, it) });
49  break;
50  }
51  else if (*it == ';') {
52  // reached a delimiter
53  paths.emplace_back(std::filesystem::path{ std::string_view(prev_begin, it) });
54  prev_begin = it + 1;
55  }
56  }
57 
58  // remove any unused space if present
59  paths.shrink_to_fit();
60  }
61 
62  return paths;
63  }
