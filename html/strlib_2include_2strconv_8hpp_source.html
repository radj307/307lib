539  * @brief Convert a floating-point number to a string, with variable precision and automatic truncation.
540  *\n This is a legacy support function.
541  * @param value: The input value.
542  * @param precision: The number of digits to set the floating-point precision field to.
543  * @param force_decimal: When true, the returned string will end with ".0" if there are no non-zero digits after the decimal point.
544  * @returns std::string
545  */
546  template<std::floating_point T>
547  [[nodiscard]] inline CONSTEXPR const std::string to_string(const T& value, const std::streamsize& precision = 8, const bool force_decimal = false)
548  {

550 
551  const size_t decimalPos{ s.rfind('.') };
552 
553  if (decimalPos < s.size()) {
