87  /**
88  * @brief Retrieve the current position of the cursor, measured in characters of the screen buffer.
89  * @tparam RT Return Type.
90  * @returns std::pair<RT, RT>
91  */
92  template<std::integral RT = size_t>
93  inline std::pair<RT, RT> getCursorPosition() noexcept(false)
94  {
95  const auto stoull{ [](auto&& str) { try { return std::stoull(std::forward<decltype(str)>(str)); } catch (...) { return 0ull; } } };
96  std::cout << ReportCursorPosition;
97  std::string row, col;
98  bool select_col{ false }; ///< @brief when true, appends digits to column; else, appends to row.
99  for (auto& c : query::getResponse()) {
100  using namespace ANSI;
101  switch (c) {
102  case '0': [[fallthrough]];
103  case '1': [[fallthrough]];
104  case '2': [[fallthrough]];
105  case '3': [[fallthrough]];
106  case '4': [[fallthrough]];
107  case '5': [[fallthrough]];
108  case '6': [[fallthrough]];
109  case '7': [[fallthrough]];
110  case '8': [[fallthrough]];
111  case '9':
112  if (!select_col)
113  row += c;
114  else
115  col += c;
116  break;
117  case ';': // delim
118  select_col = true;
119  break;
120  case '\x1b': [[fallthrough]];// ESC
121  case '\x5b': // CSI
122  break;
123  case 'R': // sequence end
124  if (!row.empty() && !col.empty()) // divider delim has been reached
125  return{ static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(row)), static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(col)) };
126  // else:
127  [[fallthrough]];
128  default:
129  throw make_exception("getCursorPosition()\tReceived unexpected character: \'", c, "\'!");
130  }
131  }
132  throw make_exception("getCursorPosition()\tDidn't receive expected escape sequence! No ending character found!");
133  }
