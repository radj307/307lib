89  /**
90  * @brief Retrieve the current position of the cursor, measured in characters of the screen buffer.
91  * @tparam RT Return Type.
92  * @returns std::pair<RT, RT>
93  */
94  template<std::integral RT = size_t>
95  inline std::pair<RT, RT> getCursorPosition() noexcept(false)
96  {
97  const auto stoull{ [](auto&& str) { try { return std::stoull(std::forward<decltype(str)>(str)); } catch (...) { return 0ull; } } };
98  std::cout << ReportCursorPosition;
99  std::string row, col;
100  bool select_col{ false }; ///< @brief when true, appends digits to column; else, appends to row.
101  for (auto& c : query::getResponse()) {
102  using namespace ANSI;
103  switch (c) {
104  case '0': [[fallthrough]];
105  case '1': [[fallthrough]];
106  case '2': [[fallthrough]];
107  case '3': [[fallthrough]];
108  case '4': [[fallthrough]];
109  case '5': [[fallthrough]];
110  case '6': [[fallthrough]];
111  case '7': [[fallthrough]];
112  case '8': [[fallthrough]];
113  case '9':
114  if (!select_col)
115  row += c;
116  else
117  col += c;
118  break;
119  case ';': // delim
120  select_col = true;
121  break;
122  case '\x1b': [[fallthrough]];// ESC
123  case '\x5b': // CSI
124  break;
125  case 'R': // sequence end
126  if (!row.empty() && !col.empty()) // divider delim has been reached
127  return{ static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(row)), static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(col)) };
128  // else:
129  [[fallthrough]];
130  default:
131  throw make_exception("getCursorPosition()\tReceived unexpected character: \'", c, "\'!");
132  }
133  }
134  throw make_exception("getCursorPosition()\tDidn't receive expected escape sequence! No ending character found!");
135  }
