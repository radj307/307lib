43  // Changes the terminal settings; disables buffered I/O, and optionally enables echo when a key is pressed.
44  void initTermiosForInstantInput(bool echo = false)
45  {
46  tcgetattr(0, &old);
47  current = old;
48  current.c_lflag &= ~ICANON; //< disable buffered I/O

50  current.c_lflag |= ECHO;

109  /**
110  * @brief Receive a query response from STDIN, and return it as a string. Any prior data waiting in STDIN is discarded.
111  * @param seqEnd - The final character in the expected ANSI sequence.
112  * @param bufSize - The minimum number of bytes to reserve in the buffer. If this is insufficient, the buffer will be expanded with a potential reallocation penalty. The default is 8.
113  * @returns The query response as a std::string
114  */
115  inline std::string getResponse(const char seqEnd, const size_t bufSize = 8) noexcept
116  {
117  // create the buffer
118  std::string buf;
119 
120  // reserve space in the buffer
121  buf.reserve(bufSize);
122 
123  // retrieve input from STDIN
124  unsigned char c;
125  bool hasSequenceStarted{ false };
126  while (kbhit()) {
127  c = static_cast<unsigned char>(getch());
128  switch (c) {
129  case '\x1b': //< ESC
130  hasSequenceStarted = true;
131  [[fallthrough]];
132  default:
133  if (hasSequenceStarted) {
134  buf += static_cast<unsigned char>(c); //< no data lost because fgetc returns 1-byte
135 
136  // break if this is the sequence end char
137  if (c == seqEnd) break;
138  }
139  }
140  if (c == seqEnd || c == EOF) break;
141  }
142 
143  // clear EOF bit
144  std::cin.clear();
145 
146  // remove unused buffer space & return
147  buf.shrink_to_fit();
148  return buf;
149  }
150  }

156  * @brief Retrieve the current position of the cursor, measured in characters of the screen buffer.
157  * @tparam RT Return Type.
158  * @returns A pair of type RT that represent the X and Y coordinates of the cursor, respectively.
159  */
160  template<std::integral RT>
161  std::pair<RT, RT> getCursorPosition() noexcept(false)
162  {
163  std::cout << ReportCursorPosition << std::flush;

165 
166  const auto& response{ query::getResponse('R', 12) };
167  if (response.empty()) throw make_exception("getCursorPosition()\tDid not receive ANSI escape sequence query response from STDIN!");
168 
169  std::string row, col;
170  for (auto& c : response) {
171  using namespace ANSI;
172  switch (c) {
173  case '0': [[fallthrough]];
174  case '1': [[fallthrough]];
175  case '2': [[fallthrough]];
176  case '3': [[fallthrough]];
177  case '4': [[fallthrough]];
178  case '5': [[fallthrough]];
179  case '6': [[fallthrough]];
180  case '7': [[fallthrough]];
181  case '8': [[fallthrough]];
182  case '9':
183  if (!select_col)
184  row += c;
185  else
186  col += c;
187  break;
188  case ';': // delim
189  select_col = true;
190  break;
191  case '\x1b': [[fallthrough]];// ESC
192  case '\x5b': // CSI
193  break;
194  case 'R': // sequence end
195  if (!row.empty() && !col.empty()) // divider delim has been reached
196  return{ static_cast<RT>(std::stoull(col)), static_cast<RT>(std::stoull(row)) };
197  // else:
198  [[fallthrough]];
199  default:
200  throw make_exception("getCursorPosition() Received unexpected character: \'", c, "\'!");
201  }
202  }
203  throw make_exception("getCursorPosition() Didn't receive expected escape sequence! No ending character found!");
204  }
