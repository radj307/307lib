35  * @brief Receive a query response from STDIN, and return it as a string.
36  *\n It is ___highly discouraged___ to call this function from outside of the term namespace.
37  * @param timeout If this many milliseconds pass without receiving any data, the function will break and return nothing.
38  * @returns std::string
39  */
40  inline std::string getResponse(const int& timeout = 256, const bool flush_output_stream = false) noexcept
41  {
42  if (flush_output_stream)
43  fflush(stdout);
44  std::string seq; // already has enough space reserved
45  #ifdef OS_WIN
46  for (int i{ 0 }; !_kbhit() && i < timeout; ++i) // wait until a "key press" or until timeout is reached.
47  std::this_thread::sleep_for(1ms);
48  while (_kbhit()) // while a key is "pressed"
49  seq += static_cast<char>(_getch()); // get key code, cast to char
50  #else
51  std::cin.clear();
52  for (char c; !std::cin.fail() && std::cin >> c; )
53  seq += c;
54  std::cin.clear();
55  #endif
56  return seq;
57  }
58  }

63  /**
64  * @brief Retrieve the current position of the cursor, measured in characters of the screen buffer.
65  * @tparam RT Return Type.
66  * @returns std::pair<RT, RT>
67  */
68  template<std::integral RT = size_t>
69  inline std::pair<RT, RT> getCursorPosition() noexcept(false)
70  {
71  const auto stoull{ [](auto&& str) { try { return std::stoull(std::forward<decltype(str)>(str)); } catch (...) { return 0ull; } } };
72  std::cout << ReportCursorPosition;
73  std::string row, col;
74  bool select_col{ false }; ///< @brief when true, appends digits to column; else, appends to row.
75  for (auto& c : query::getResponse()) {
76  using namespace ANSI;
77  switch (c) {
78  case '0': [[fallthrough]];
79  case '1': [[fallthrough]];
80  case '2': [[fallthrough]];
81  case '3': [[fallthrough]];
82  case '4': [[fallthrough]];
83  case '5': [[fallthrough]];
84  case '6': [[fallthrough]];
85  case '7': [[fallthrough]];
86  case '8': [[fallthrough]];
87  case '9':
88  if (!select_col)
89  row += c;
90  else
91  col += c;
92  break;
93  case ';': // delim
94  select_col = true;
95  break;
96  case '\x1b': [[fallthrough]];// ESC
97  case '\x5b': // CSI
98  break;
99  case 'R': // sequence end
100  if (!row.empty() && !col.empty()) // divider delim has been reached
101  return{ static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(row)), static_cast<RT>(!!_internal::CURSOR_MIN_AXIS + stoull(col)) };
102  // else:
103  [[fallthrough]];
104  default:
105  throw make_exception("getCursorPosition()\tReceived unexpected character: \'", c, "\'!");
106  }
107  }
108  throw make_exception("getCursorPosition()\tDidn't receive expected escape sequence! No ending character found!");
109  }
