201  * @param lvl Input Log Level. Must be equal to 1, or a multiple of 2 to be valid.

204  inline level::LogLevel include_all_below(level::LogLevel lvl)
205  {
206  if (lvl != 1 && lvl % 2 != 0)
207  return lvl;

398  /**
399  * @brief Stream insertion operator for the xLogs class.
400  *\n Allows using xLogs in the same way as a std::ostream, with some important changes:
401  *\n - A message level must be specified before messages can be inserted, unless prefixes are disabled.
402  *\n - Messages are terminated with `endm` / nullptr.
403  *\n - Messages are only sent to the output target once endm has been received.
404  * @tparam T Input Type
405  * @param oxl (implicit) xLogs instance.
406  * @param m (implicit) Message type.
407  */
408  template<var::Streamable T>
409  friend xLogs<StreamType>& operator<<(xLogs<StreamType>& oxl, const T& m)
410  {
411  if constexpr (std::same_as<T, level::LogLevel>)
412  oxl._last = std::make_unique<level::LogLevel>(m);
413  else if constexpr (std::same_as<T, term::Message>)
414  oxl._last = std::make_unique<level::LogLevel>(message_to_level(m));
415  else if constexpr (std::same_as<T, msg_break_t>) {
416  if (!oxl._add_prefix || oxl._last.get() == nullptr)
417  oxl.write(level::NONE, oxl._buffer.str());
418  else
419  oxl.write(*oxl._last.get(), oxl._buffer.str());
420  oxl._buffer = {};
421  }
422  else if (!oxl._add_prefix || oxl._last.get() != nullptr)
423  oxl._buffer << m;
424  return oxl;
425  }

428  * @param oxl (implicit) xLogs instance.
