373  /**
374  * @brief Parse commandline arguments into an ArgContainer instance.
375  *\n __Argument Types__
376  *\n - Parameters are any arguments that do not begin with a dash '-' character that were not captured by another argument type.
377  *\n - Options are arguments that begin with 2 dash '-' characters, and can capture additional arguments if the option name appears in the capture list.
378  *\n - Flags are arguments that begin with a single dash '-' character, are a single character in length, and can capture additional arguments. Flags can appear alone, or in "chains" where each character is treated as an individual flag. In a flag chain, only the last flag can capture additional arguments.
379  *\n __Capture Rules__
380  *\n - Only options/flags specified in the capture list are allowed to capture additional arguments. Capture list entries should not include a delimiter prefix.
381  *\n - Options/Flags cannot be captured under any circumstance. ex: "--opt --opt captured" results in "--opt", & "--opt" + "captured".
382  *\n - If a flag in a chain should capture an argument (either with an '=' delimiter or by context), it must appear at the end of the chain.
383  *\n - Any captured arguments do not appear in the argument list by themselves, and must be accessed through the argument that captured them.
384  * @param args Commandline arguments as a vector of strings, in order and including argv[0].
385  * @param captures A CaptureList instance specifying which arguments are allowed to capture other arguments as their parameters
386  * @returns ArgContainer
387  */
388  inline ArgContainerType parse(StrVec&& args, const CaptureList& captures)
389  {
390  // remove empty arguments, which are possible when passing arguments from automated testing applications
391  args.erase(std::remove_if(args.begin(), args.end(), [](auto&& s) { return s.empty(); }), args.end());
392 
393  ArgContainerType cont{};
394  cont.reserve(args.size());
395 
396  for (StrVec::const_iterator it{ args.begin() }; it != args.end(); ++it) {
397  auto [arg, d_count] { captures.parsingRules.stripPrefix(*it, 2ull) };
398 
399  switch (d_count) {
400  case 2ull: // Option
401  if (const auto eqPos{ arg.find('=') }; eqPos != std::string::npos) {// argument contains an equals sign
402  auto opt{ arg.substr(0ull, eqPos) }, cap{ arg.substr(eqPos + 1ull) };
403 
404  if (const auto& captureStyle{ captures.get_capture_style(opt) }; !captureStyle.isDisabled())
405  cont.emplace_back(opt::Option(std::make_pair(std::move(opt), std::move(cap))));
406  else {
407  if (captureStyle.isRequired())
408  throw make_exception("Expected a capture argument for option '", opt, "'!");
409  cont.emplace_back(opt::Option(std::make_pair(std::move(opt), std::nullopt)));
410  if (!cap.empty()) {
411  arg = cap;
412  goto JUMP_TO_PARAMETER; // skip flag case, add invalid capture as a parameter
413  }
414  }
415  }
416  else if (const auto& captureStyle{ captures.get_capture_style(arg) }; !captureStyle.isDisabledOrEqualsOnly() && captures.parsingRules.canCaptureNext(it, args.end())) // argument can capture next arg
417  cont.emplace_back(opt::Option(std::make_pair(arg, *++it)));
418  else {
419  if (captureStyle.isRequired())
420  throw make_exception("Expected a capture argument for option '", arg, "'!");
421  cont.emplace_back(opt::Option(std::make_pair(arg, std::nullopt)));
422  }
423  break;
424  case 1ull: // Flag
425  if (!captures.parsingRules.isNumber(arg)) { // single-dash prefix is not a number
426  std::optional<opt::Flag> capt{ std::nullopt }; // this can contain a flag if there is a capturing flag at the end of a chain
427  std::string invCap{}; //< for invalid captures that should be treated as parameters
428  if (const auto eqPos{ arg.find('=') }; eqPos != std::string::npos) {
429  invCap = arg.substr(eqPos + 1ull); // get string following '=', use invCap in case flag can't capture
430  if (const auto flag{ arg.substr(eqPos - 1ull, 1ull) }; !captures.get_capture_style(flag).isDisabled()) {
431  capt = opt::Flag(flag.front(), invCap); // push the capturing flag to capt, insert into vector once all other flags in this chain are parsed
432  arg = arg.substr(0ull, eqPos - 1ull); // remove last flag, '=', and captured string from arg
433  invCap.clear(); // flag can capture, clear invCap
434  }
435  else
436  arg = arg.substr(0ull, eqPos); // remove everything from eqPos to arg.end()
437  }
438  // iterate through characters in arg
439  for (auto fl{ arg.begin() }; fl != arg.end(); ++fl) {
440  const auto& captureStyle{ captures.get_capture_style(std::string(1ull, *fl)) };
441  // If this is the last char, and it can capture
442  if (fl == arg.end() - 1ll && !captureStyle.isDisabledOrEqualsOnly() && captures.parsingRules.canCaptureNext(it, args.end()))
443  cont.emplace_back(opt::Flag(std::make_pair(*fl, *++it)));
444  else {// not last char, or can't capture
445  if (captureStyle.isRequired())
446  throw make_exception("Expected a capture argument for flag '", *fl, "'!");
447  cont.emplace_back(opt::Flag(std::make_pair(*fl, std::nullopt)));
448  }
449  }
450  if (capt.has_value()) // flag captures are always at the end, but parsing them first puts them out of chronological order.
451  cont.emplace_back(std::move(capt.value()));
452  if (invCap.empty())
453  break;
454  else arg = invCap; // set argument to invalid capture and fallthrough to add it as a parameter
455  }
456  else // this is a negative number, re-add '-' prefix and fallthrough
457  arg = *it;
458  [[fallthrough]];
459  case 0ull:
460  JUMP_TO_PARAMETER:
461  [[fallthrough]]; // Parameter
462  default:
463  cont.emplace_back(opt::Parameter(arg));
464  break;
465  }
466  }
467  cont.shrink_to_fit();
468 
469  // count the number of each included argument:
470  std::unordered_map<std::string, size_t> counts;
471 
472  for (const auto& varg : cont) {
473  std::visit([&counts](auto&& value) {
474  using T = std::decay_t<decltype(value)>;
475 
476  if constexpr (std::same_as<T, Option>)
477  ++counts[value.name()];
478  else if constexpr (std::same_as<T, Flag>)
479  ++counts[value.name()];
480  }, varg);
481  }
482 
483  // validate argument count limits:
484  for (const auto& it : captures.vec) {
485  const auto& name{ it.name() };
486  if (const auto& count{ counts[name] }; !it.withinRequiredRange(count)) {
487  throw make_exception("Argument count ", count, " for '", name, "' is out-of-bounds! Expected (min: ", it.minCount, ", max: ", ([&it]() {
488  if (it.maxCount.has_value())
489  return std::to_string(it.maxCount.value());
490  return "(none)"s;
491  }()), ')');
492  }
493  }
494  return cont;
