388  * @tparam ReturnType Type of integral to return.
389  * @param hex_str Hexadecimal value as a string.
390  *\n Note that this CANNOT contain prefix characters like '#' or "0x"
391  * @returns ReturnType

393  template<std::integral ReturnType>
394  inline static CONSTEXPR ReturnType hex_to_int(const std::string& hex_str) noexcept
395  {
396  if (hex_str.empty())
397  return static_cast<ReturnType>(0);
398  const auto base{ static_cast<ReturnType>(16) };
399  ReturnType power{ 1 }, result{ 0 };
400  for (auto ch{ hex_str.rbegin() }; ch != hex_str.rend(); ++ch) {
401  if (const auto v{ hex_to_int<ReturnType>(*ch) }; v < base) {
402  result += v * power;
403  power = power * base;
404  }
