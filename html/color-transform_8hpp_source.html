152  {

154  return static_cast<ReturnType>(0);
155  const auto base{ static_cast<ReturnType>(16) };
156  ReturnType power{ 1 }, result{ 0 };
157  for (auto ch{ hex_str.rbegin() }; ch != hex_str.rend(); ++ch) {
158  if (const auto v{ hex_to_int<ReturnType>(*ch) }; v < base) {
159  result += v * power;
