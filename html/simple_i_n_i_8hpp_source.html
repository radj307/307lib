533  /// @returns true when the mask contains the given header, or the mask is empty *(it wasn't specified)*; otherwise false.

549  CONSTEXPR bool _$isCommentChar(const char c) const noexcept

553  /// @returns A string_view of the parameter 'line' that excludes any line comments, as determined by the commentStyle & some mild parsing.
554  WINCONSTEXPR std::string_view _$stripComments(std::string const& line) const noexcept
555  {
556  bool escaped{ false }, single_quoted{ false }, double_quoted{ false };
557  for (size_t i{ 0 }, end{ line.size() }; i < end; ++i) {
558  const char& c{ line.at(i) };
559 
560  if (escaped)
561  escaped = false;
562  else if (_$isEscapeChar(c))
563  escaped = true;
564  else if (c == '\'')
565  single_quoted = !single_quoted;
566  else if (c == '\"')
567  double_quoted = !double_quoted;
568  else if (!single_quoted && !double_quoted && _$isCommentChar(c))
569  return std::string_view{ line.begin(), line.begin() + i };
570  // else continue
571  }
572  return std::string_view{ line };
573  }
574  /// @returns A pair of iterators pointing to enclosing *(must be matching & both unescaped)* quotation marks in the given string. If no valid enclosing quotes are found, returns { value.end(), value.end() } instead.
575  WINCONSTEXPR std::pair<typename std::string::const_iterator, typename std::string::const_iterator> _$findEnclosingQuotes(std::string const& value) const noexcept
576  {
577  size_t i;
578  char delim;
579  const auto& findDelim{ [&value, &i, &delim, this](auto&& ch) {
580  const auto& idx{ i++ };
581  return ($fwd(ch) == delim && (idx == 0 || (idx > 0 && !_$isEscapeChar(value.at(idx)))));
582  } };
583 
584  // find double quotes
585  i = 0ull;
586  delim = '\"';
587  if (const auto& fst{ std::find_if(value.begin(), value.end(), findDelim) }; fst != value.end()) {
588  if (const auto& snd{ std::find_if(fst + 1, value.end(), findDelim) }; snd != value.end()) {
589  return{ fst, snd };
590  }
591  }
592 
593  // find single quotes
594  i = 0ull;
595  delim = '\'';
596  if (const auto& fst{ std::find_if(value.begin(), value.end(), findDelim) }; fst != value.end()) {
597  if (const auto& snd{ std::find_if(fst + 1, value.end(), findDelim) }; snd != value.end()) {
598  return{ fst, snd };
599  }
600  }
601  return{ value.end(), value.end() };
602  }

613 #pragma region deep_merge
614  /**
615  * @brief Performs a recursive merge of two given INI containers.
616  * @param container Input container reference.
617  * @param other Other container constant reference.
618  * @param overrideStyle The header/key override style to use for duplicate headers/keys.
619  * @returns The reference of the container parameter, with all of the sections from other merged into it.
620  */
621  template<class TKeyComparator = CaseInsensitiveCompare>
622  ini_container<TKeyComparator>& deep_merge(ini_container<TKeyComparator>& container, ini_container<TKeyComparator> const& other, OverrideStyle const& overrideStyle = OverrideStyle::Override)
623  {
624  for (const auto& [header, section] : other) {
625  if (const auto& existing{ container.find(header) }; existing != container.end()) {
626  auto& [existingHeader, existingSection] { *existing };
627  for (const auto& [key, value] : section) {
628  if (existingSection.contains(key)) {
629  switch (overrideStyle) {
630  case OverrideStyle::OnlyBlank:
631  if (!section.at(key).empty())
632  break; //< break if current value is NOT empty
633  else [[fallthrough]];
634  case OverrideStyle::Override:
635  existingSection.at(key) = value;
636  break;
637  case OverrideStyle::Throw:
638  throw ex::make_custom_exception<ini_key_exception>("Duplicate keys aren't allowed; '", header, (header.empty() ? "" : "::"), key, "' already has value '", section.at(key), "'! (Incoming value: '", value, "')");
639  case OverrideStyle::Skip: [[fallthrough]];
640  default:break;
641  }
642  }
643  else existingSection[key] = value;
644  }
645  }
646  else container.insert(std::make_pair(header, section));
647  }
648  return container;
649  }

652 #pragma region parse
653  /**
654  * @brief Parses the given input stream into an INI container object.
655  * @param is An input stream to parse data from.
656  * @param config Optional configuration object that changes the behaviour of the parser.
657  * @returns An ini_container type that contains all headers, keys, and values from the input stream.
658  * @throws ini_syntax_exception Input stream contains invalid syntax, and the syntaxErrorStyle specified by the config was SyntaxErrorStyle::Throw
659  * @throws ini_key_exception Input stream contains duplicate keys, and the overrideStyle specified by the config was OverrideStyle::Throw
660  */
661  template<class TKeyComparator = CaseInsensitiveCompare>
662  INLINE ini_container<TKeyComparator> parse(std::istream& is, ini_parser_config<TKeyComparator> const& config = {}) noexcept(false)
663  {
664  static const auto WHITESPACE{ " \t\v\r\n" };
665 
666  ini_container<TKeyComparator> ini{};
667 
668  size_t ln{ 0 };
669 
670  std::string header{};
671  bool skip_header{ false };
672 
673  for (std::string line; std::getline(is, line, '\n'); is.clear(), ++ln) {
674  if (std::all_of(line.begin(), line.end(), str::stdpred::isspace)) continue;
675 
676  auto l{ config._$stripComments(line) };
677 
678  if (std::all_of(l.begin(), l.end(), str::stdpred::isspace)) continue;
679 
680  // find headers
681  if (const size_t iOpen{ l.find($c(char, '[')) }, iClose{ l.rfind($c(char, ']')) };
682  iOpen != std::string::npos && iClose != std::string::npos) {
683  std::string tmpHeader{ l.substr(iOpen + 1, iClose - iOpen - 1) };
684 
685  if (config.stripWhitespaceFromHeaders)
686  tmpHeader = str::trim(tmpHeader);
687 
688  if (const size_t fstNonSpace{ l.find_first_not_of(WHITESPACE) }, lastNonSpace{ l.find_last_not_of(WHITESPACE) };
689  (iOpen > fstNonSpace || iClose < lastNonSpace) && config.syntaxErrorStyle == SyntaxErrorStyle::Throw)
690  throw ex::make_custom_exception<ini_syntax_exception>("Line ", ln, " contains a header with preceding or trailing non-whitespace characters! '", line, "'");
691 
692  skip_header = false;
693 
694  if (!config._$maskIncludes(tmpHeader)) {
695  switch (config.maskStyle) {
696  case MaskStyle::Skip:
697  skip_header = true;
698  break;
699  case MaskStyle::Throw:
700  throw ex::make_custom_exception<ini_mask_exception>("Line ", ln, " contains an unexpected header: '", tmpHeader, "'!");
701  case MaskStyle::Disable: [[fallthrough]];
702  default:break;
703  }
704  }
705 
706  header = tmpHeader;
707  }
708  else if (!skip_header) {
709  // find key-value pairs
710  if (const size_t equals{ l.find('=') }; equals != std::string::npos) {
711  if (const std::string key{ str::trim(l.substr(0ull, equals)) }; !key.empty()) {
712  auto& section{ ini[header] };
713 
714  std::string value{ l.substr(equals + 1).data() };
715 
716  if (config.stripWhitespaceFromValue)
717  value = str::trim(value); //< remove unenclosed preceding/trailing whitespace
718 
719  // find enclosing quotes
720  if (const auto& [fst, snd] { config._$findEnclosingQuotes(value) };
721  fst != value.end() && snd != value.end()) {
722  // validate the line if syntax errors are enabled
723  if (config.syntaxErrorStyle == SyntaxErrorStyle::Throw) {
724  if (fst != value.begin() && !std::all_of<typename std::string::const_iterator>(value.begin(), fst, str::stdpred::isspace))
725  throw ex::make_custom_exception<ini_syntax_exception>("Line ", ln, " the setter for key '", key, "' specifies a quote-enclosed value, but there were unexpected characters before the opening quote: {", std::string_view{ value.begin(), fst }, "}!");
726  else if (snd != value.end() && !std::all_of<typename std::string::const_iterator>(snd + 1, value.end(), str::stdpred::isspace))
727  throw ex::make_custom_exception<ini_syntax_exception>("Line ", ln, " the setter for key '", key, "' specifies a quote-enclosed value, but there were unexpected characters after the closing quote: {", std::string_view{ snd + 1, value.end() }, "}!");
728  }
729 
730  // update the value; optionally include quotes when the config enables it
731  value = { fst + config.stripEnclosingQuotes, snd + !config.stripEnclosingQuotes };
732  }
733 
734  if (!config._$maskIncludes(header, key)) {
735  switch (config.maskStyle) {
736  case MaskStyle::Skip:
737  continue; //< skip key
738  case MaskStyle::Throw:
739  throw ex::make_custom_exception<ini_mask_exception>("Line ", ln, " contains unexpected key '", key, "'", (header.empty() ? std::string{ " within section '"s + header + "'!" } : "!"s));
740  case MaskStyle::Disable: [[fallthrough]];
741  default: break;
742  }
743  }
744 
745  if (section.contains(key)) {
746  switch (config.overrideStyle) {
747  case OverrideStyle::OnlyBlank:
748  if (!section.at(key).empty())
749  break; //< break if current value is NOT empty
750  else [[fallthrough]];
751  case OverrideStyle::Override:
752  section.at(key) = value;
753  break;
754  case OverrideStyle::Throw:
755  throw ex::make_custom_exception<ini_key_exception>("Line ", ln, " specifies a duplicate key '", value, "'; '", header, (header.empty() ? "" : "::"), key, "' already has value '", section.at(key), "'!");
756  case OverrideStyle::Skip: [[fallthrough]];
757  default:break;
758  }
759  }
760  else section.insert(std::make_pair(key, value));
761  }
762  else if (config.syntaxErrorStyle == SyntaxErrorStyle::Throw)
763  throw ex::make_custom_exception<ini_syntax_exception>("Invalid key specified on line ", ln, ": \"", line, '\"');
764  }
765  }
766  }
767 
768  if (config.addMaskToOutput)
769  deep_merge<TKeyComparator>(ini, config.mask, OverrideStyle::OnlyBlank);
770 
771  return ini;
772  }
