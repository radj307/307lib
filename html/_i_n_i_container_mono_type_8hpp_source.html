37  /**
38  * @brief Convert a T to a std::string
39  * @param var - Input variable
40  * @param use_quotes_for_string - When true, variables with String type are returned with enclosing double quotation marks for file/human-readable output.
41  * @returns std::string
42  * @throws bad_variant_access when the given T instance is null/std::monostate.
43  */
44  template<ValidValueT T>
45  std::string to_string(const T& var, const bool use_quotes_for_string = true)
46  {
47  if constexpr (std::same_as<T, String>) // String
48  return (use_quotes_for_string ? "\"" + var + "\"" : var);
49  else if constexpr (std::same_as<T, Boolean>) // Boolean
50  return str::bool_to_string(var);
51  return std::to_string(var); // other

114  /**
115  * @brief Stream insertion operator. Outputs in valid file format, with quotation-enclosed strings.
116  * @param os - (implicit) Target output stream.
117  * @param obj - (implicit) Target INIContainer instance.
118  * @returns std::ostream&
119  */
120  friend std::ostream& operator<<(std::ostream& os, const INIContainerMonoType& obj)
121  {
122  using enum AlignType;
123  std::streamsize align_width{ 0ll };
124  switch (obj._align_output) {
125  case SECTION:
126  for (const auto& [header, section] : obj._cont) {
127  if (!header.empty())
128  os << '[' << header << ']' << '\n';
129  const auto align_width{ [&section]() -> std::streamsize {
130  std::streamsize longest{ 0ll };
131  for (auto& [keyname, _] : section)
132  if (const auto sz{ keyname.size() }; sz > longest)
133  longest = sz;
134  return longest;
135  }() };
136  for (auto& [key, val] : section)
137  os << key << str::VIndent(align_width + 2ll, key.size()) << "= " << to_string(val) << '\n';
138  os << '\n';
139  }
140  break;
141  case ALL:
142  align_width = [&obj]() -> std::streamsize {
143  std::streamsize longest{ 0ll };
144  for (auto& [_, section] : obj._cont)
145  for (auto& [keyname, __] : section)
146  if (const auto sz{ keyname.size() }; sz > longest)
147  longest = sz;
148  return longest;
149  }();
150  [[fallthrough]];
151  case NONE:
152  for (auto& [header, section] : obj._cont) {
153  os << '[' << header << ']' << '\n';
154  for (auto& [key, value] : section)
155  os << key << str::VIndent(align_width + 2ll, key.size()) << "= " << to_string(value, true) << '\n';
156  os << '\n';
157  }
158  break;
159  }
160  return os;
161  }
