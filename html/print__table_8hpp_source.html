143  {
144  switch (alignment) {
145  case HorizontalAlignment::Left:
146  os << indent(padding) << str << indent(padding) << indent(width, str.size() + padding * 2);
147  break;
148  case HorizontalAlignment::Center: {
149  const auto total_indentsz{ width - str.size() - padding * 2ull }, indentsz{ total_indentsz / 2 };
150  os
