132  {
133  if (isdigit(ch))

142  /**
143  * @brief Convert a single hexadecimal value to an integral.
144  * @tparam ReturnType Type of integral to return.
145  * @param hex_str Hexadecimal value as a string.
146  *\n Note that this CANNOT contain prefix characters like '#' or "0x"
147  * @returns ReturnType
148  */
149  template<::std::integral ReturnType>
150  inline static _CONSTEXPR ReturnType hex_to_int(const std::string& hex_str) noexcept
151  {
152  if (hex_str.empty())
153  return static_cast<ReturnType>(0);
154  const auto base{ static_cast<ReturnType>(16) };
155  ReturnType power{ 1 }, result{ 0 };
156  for (auto ch{ hex_str.rbegin() }; ch != hex_str.rend(); ++ch) {
157  if (const auto v{ hex_to_int<ReturnType>(*ch) }; v < base) {
158  result += v * power;
159  power = power * base;
160  }
161  }
162  return result;
163  }
