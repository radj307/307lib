42  {
43  // create a stack to enumerate the tree depth-first
44  std::stack<std::tuple<std::string, T, size_t, size_t>> stack;
45 
46  // push the root node onto the stack
47  stack.push(std::make_tuple("", p.root_object, 0ull, 0ull));
48 
49  while (!stack.empty()) {
50  // take the next node off of the stack
51  auto [prefix, node, depth, linePositions] { stack.top() };
52  stack.pop();
53 
54  // print this node with the previously-calculated prefix
55  os << prefix;
56  p.print_node(os, node);
57  os << '\n';
58 
59  //< nothing is printed to (os) beyond this line
60 
61  if (const auto& children{ p.select_child_nodes(node) }; !children.empty()) {
62  // enumerate this node's children & create a string that contains any
63  // prefix box characters to maintain the tree structure's appearance
64  auto indentSize{ depth * p.indent_size };
65 
66  for (auto it{ children.rbegin() }, it_end{ children.rend() }, it_last{ it }; it != it_end; ++it) {
67  std::stringstream ss;
68 
69  // enable DEC line drawing characters
70  ss << term::EnableLineDrawing;
71 
72  // print indentation
73  if (indentSize > 0) {
74  for (auto i{ 0 }; i < indentSize; i += p.indent_size) {
75  if ((linePositions & (1ull << (i / p.indent_size))) != 0) {
76  // a vertical line for a previously-printed node is required here
77  ss << term::LineCharacter::LINE_VERTICAL;
78  }
79  else ss << ' ';
80  // fill in the rest of the indentation
81  if (p.indent_size > 1)
82  ss << indent(p.indent_size - 1);
83  }
84  }
85 
86  // print tree structure line drawings for this child node
87  if (it == it_last) {
88  // this is the last of this node's children to be printed:
89  ss << term::LineCharacter::CORNER_BOTTOM_LEFT;
90  }
91  else {
92  // there are still more of this node's children to print:
93  ss << term::LineCharacter::JUNCTION_3_WAY_LEFT;
94  // make sure any sub-nodes of this child print a vertical bar at this
95  // node's depth in order to maintain the tree structure's appearance:
96  linePositions |= (1ull << depth);
97  }
98  // print horizontal lines until we reach the value position
99  for (std::uint16_t i{ 0 }, i_max{ p.indent_size - 1 }; i < i_max; ++i) {
100  ss << term::LineCharacter::LINE_HORIZONTAL;
101  }
102 
103  // disable DEC line drawing characters
104  ss << term::DisableLineDrawing;
105 
106  stack.push(std::make_tuple(ss.str(), *it, depth + 1, linePositions));
107  }
